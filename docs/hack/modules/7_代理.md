## 修改点 2

### 修改后
```ts
constructor(/* .... */){
	// NOTE - 配置修改后，刷新
	this._register(vscode.workspace.onDidChangeConfiguration(e => {
		if (e.affectsConfiguration('github.copilot.hackModels')) {
			this._onDidChange.fire();
		}
	}));
}

public async getModels(): Promise<CopilotCLIModelInfo[]> {
	if (!this._authenticationService.anyGitHubSession) {
		return [];
	}

	// No need to query sdk multiple times, cache the result, this cannot change during a vscode session.
	return this._getAvailableModels();
}

private async _getAvailableModels(): Promise<CopilotCLIModelInfo[]> {
	// const [{ getAvailableModels }, authInfo] = await Promise.all([this.copilotCLISDK.getPackage(), this.copilotCLISDK.getAuthInfo()]);
	try {

		const modelConfigs = vscode.workspace.getConfiguration('github.copilot.hackModels').get('extras', []) as IModelAPIResponse[];

		const models = modelConfigs.map(model => {
			const config = model.capabilities as IChatModelCapabilities;
			// 为 limits 设置默认值
			if (config.limits == undefined) {
				config.limits = {
					max_context_window_tokens: 128000,
					max_output_tokens: 32000,
					max_prompt_tokens: 64000
				};
			} else {
				// 为已存在的 limits 设置默认值
				config.limits.max_context_window_tokens = config.limits.max_context_window_tokens ?? 128000;
				config.limits.max_output_tokens = config.limits.max_output_tokens ?? Math.floor(config.limits.max_context_window_tokens * 0.25);
				config.limits.max_prompt_tokens = config.limits.max_prompt_tokens ?? Math.floor(config.limits.max_context_window_tokens * 0.5);
			}

			return {
					id: model.model,
					name: model.model,
				maxContextWindowTokens: config.limits?.max_context_window_tokens,
				maxInputTokens: config.limits?.max_prompt_tokens,
				maxOutputTokens: config.limits?.max_output_tokens,
				supportsVision: config.supports.vision
			} as CopilotCLIModelInfo;
		})

		const base = vscode.workspace.getConfiguration('github.copilot.hackModels.base');
		if (base && base.get("model_picker_enabled", true) === true) {

			const config: IChatModelCapabilities = {
				type: "chat",
				family: "gpt-5-mini",
				tokenizer: base.get("capabilities.tokenizer", TokenizerType.O200K),
				limits: {
					max_context_window_tokens: base.get("capabilities.limits.max_context_window_tokens", 128000),
					max_output_tokens: base.get("capabilities.limits.max_output_tokens", Math.floor(base.get("capabilities.limits.max_context_window_tokens", 128000) * 0.25)),
					max_prompt_tokens: base.get("capabilities.limits.max_prompt_tokens", Math.floor(base.get("capabilities.limits.max_context_window_tokens", 128000) * 0.5)),
					vision: {
						max_prompt_images: base.get("capabilities.limits.vision.max_prompt_images", 1)
					}
				},
				supports: {
					vision: base.get("capabilities.supports.vision", true),
					streaming: base.get("capabilities.supports.streaming", true),
				},
			} as IChatModelCapabilities;



			models.push({
				id: base.get("model", "gpt-5.2"),
				name: base.get("model", "gpt-5.2"),
				maxContextWindowTokens: config.limits?.max_context_window_tokens,
				maxInputTokens: config.limits?.max_prompt_tokens,
				maxOutputTokens: config.limits?.max_output_tokens,
				supportsVision: config.supports.vision
			} as CopilotCLIModelInfo);

		}

		return models;
	} catch (ex) {
		this.logService.error(`[CopilotCLISession] Failed to fetch models`, ex);
		return [];
	}
}
```

### 说明
从 `github.copilot.hackModels` 配置读取模型，包括 extras 和 base 配置。

## 修改点 3
- **文件**: `src\extension\chatSessions\copilotcli\node\copilotcliSessionService.ts`
- **位置**: `createSessionsOptions` 方法内部，`allOptions.model = options.model` 之后

### 修改后
```ts
let modelConfig: IModelAPIResponse | undefined = undefined;

if (options.model) {
	const models = vscode.workspace.getConfiguration('github.copilot.hackModels').get('extras', []) as IModelAPIResponse[];
	for (let item of models) {
		if (item.model === options.model) {
			item.supported_endpoints = item.supported_endpoints ? item.supported_endpoints : [ModelSupportedEndpoint.ChatCompletions];
			modelConfig = item;
			break;
		}
	}
}
if (modelConfig == undefined) {
	const base_config = vscode.workspace.getConfiguration('github.copilot.hackModels.base');
	modelConfig = {
		id: "gpt-5-mini",
		preview: false,
		baseUrl: base_config.get("baseUrl", "http://steam"),
		apiKey: base_config.get("apiKey", "xxxxx"),
		name: base_config.get("model", "gpt-5.2"),
		model: base_config.get("model", "gpt-5.2"),
		model_picker_enabled: true,
		version: base_config.get("version", "v1.0.0"),
		supported_endpoints: base_config.get("supported_endpoints", [ModelSupportedEndpoint.ChatCompletions]) as ModelSupportedEndpoint[],
		capabilities: {
			type: "chat",
			family: "gpt-5-mini",
			tokenizer: base_config.get("capabilities.tokenizer", TokenizerType.O200K),
			supports: {
				streaming: base_config.get("capabilities.supports.streaming", true),
			},
		}
	} as IModelAPIResponse;
}

let apiType: "openai" | "anthropic" = 'openai';
let wireApi: "completions" | "responses" | undefined = 'completions';
if (modelConfig) {
	allOptions.model = modelConfig.model;

	if (modelConfig.supported_endpoints && modelConfig.supported_endpoints.length > 0) {
		switch (modelConfig.supported_endpoints[0]) {
			case ModelSupportedEndpoint.ChatCompletions: apiType = 'openai'; wireApi = "completions"; break;
			case ModelSupportedEndpoint.Responses: apiType = 'openai'; wireApi = "responses"; break;
			case ModelSupportedEndpoint.Messages:
						apiType = 'anthropic';
						wireApi = undefined;

						if (modelConfig.baseUrl && modelConfig.baseUrl.toLowerCase().endsWith("/v1")) {
							modelConfig.baseUrl = modelConfig.baseUrl.substring(0, modelConfig.baseUrl.length - 3);
						}

						break;
			default:
				break;
		}
	}
}

allOptions.provider = {
	type: apiType,
	apiKey: modelConfig?.apiKey,
	baseUrl: modelConfig?.baseUrl ?? "",
	wireApi: wireApi,
};
```

### 说明
在 `createSessionsOptions` 方法中，从 `github.copilot.hackModels.extras` 配置读取模型，如果未找到则使用 `github.copilot.hackModels.base` 配置。根据 `supported_endpoints` 设置 API 类型（openai/anthropic）和 wireApi（completions/responses/undefined）。
