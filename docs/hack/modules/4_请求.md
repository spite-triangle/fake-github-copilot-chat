### 辅助函数 getModelConfig

/**
 * 根据模型名称获取配置
 * @param model 模型名称
 * @returns 配置对象 { baseUrl, apiKey, model }
 */
function getModelConfig(model: string | undefined): { baseUrl: string; apiKey: string; model: string } | undefined {
	if (!model) return undefined;

	const models = workspace.getConfiguration('github.copilot.hackModels');
	const extras = models.get('extras', []) as Array<{ model?: string; baseUrl?: string; apiKey?: string }>;

	// 模型映射规则
	if (model === 'gpt-5-mini') {
		const config = workspace.getConfiguration('github.copilot.hackModels.base');
		return {
			baseUrl: config.get('baseUrl', ''),
			apiKey: config.get('apiKey', ''),
			model: config.get('model', model),
		};
	} else if (model === 'text-embedding-3-small') {
		const config = workspace.getConfiguration('github.copilot.hackModels.embedding');
		return {
			baseUrl: config.get('baseUrl', ''),
			apiKey: config.get('apiKey', ''),
			model: config.get('model', model),
		};
	} else if (model === 'gpt-4.1' || model === 'gpt-4o-mini') {
		const config = workspace.getConfiguration('github.copilot.hackModels.fast');
		return {
			baseUrl: config.get('baseUrl', ''),
			apiKey: config.get('apiKey', ''),
			model: config.get('model', model),
		};
	} else {
		// 在 extras 配置中查找
		for (const extra of extras) {
			if (extra.model === model) {
				return {
					baseUrl: extra.baseUrl ?? '',
					apiKey: extra.apiKey ?? '',
					model: extra.model ?? model,
				};
			}
		}
	}
	return undefined;
}
```

### networkRequest 函数重写（完整代码）

```ts
function networkRequest(
	accessor: ServicesAccessor,
	options: INetworkRequestOptions,
): Promise<Response> {
	const fetcher = accessor.get(IFetcherService);
	const telemetryService = accessor.get(ITelemetryService);
	const capiClientService = accessor.get(ICAPIClientService);
	const { requestType, endpointOrUrl, secretKey, intent, requestId, body, additionalHeaders, cancelToken, useFetcher, canRetryOnce = true, location } = options;

	// TODO @lramos15 Eventually don't even construct this fake endpoint object.
	const endpoint = typeof endpointOrUrl === 'string' || 'type' in endpointOrUrl ? {
		modelMaxPromptTokens: 0,
		urlOrRequestMetadata: endpointOrUrl,
		family: '',
		tokenizer: TokenizerType.O200K,
		acquireTokenizer: () => {
			throw new Error('Method not implemented.');
		},
		name: '',
		version: '',
	} satisfies IEndpoint : endpointOrUrl;
	const agentInteractionType = options.requestKindOptions?.kind === 'subagent' ?
		'conversation-subagent' :
		options.requestKindOptions?.kind === 'background' ?
			'conversation-background' :
			intent === 'conversation-agent' ? intent :
				intent;

	let url: string | undefined = undefined;
	let apiKey = secretKey;

	// NOTE - 根据模型配置获取 baseUrl 和 apiKey
	if (body?.model) {
		const modelConfig = getModelConfig(body.model);
		if (modelConfig) {
			apiKey = modelConfig.apiKey;
			url = modelConfig.baseUrl;
			body.model = modelConfig.model;

			// 根据 RequestType 设置 URL 路径
			if (typeof endpoint.urlOrRequestMetadata !== 'string' && 'type' in endpoint.urlOrRequestMetadata) {
				switch (endpoint.urlOrRequestMetadata.type) {
					case RequestType.Models: url += "/models"; break;
					case RequestType.CAPIEmbeddings: url += "/embeddings"; break;
					case RequestType.ChatCompletions:
						url += "/chat/completions";
						const models = workspace.getConfiguration("github.copilot.hackModels");
						body.frequency_penalty = models.get("frequency_penalty", undefined);
						body.presence_penalty = models.get("presence_penalty", undefined);
						break;
					case RequestType.ChatResponses: url += "/responses"; break;
					case RequestType.ChatMessages: url += "/messages"; break;
					default:
						url = undefined;
				}
			}
		}
	}

	// 如果没有通过上述方式获取 url，尝试使用 Proxy 配置
	if (!url && typeof endpoint.urlOrRequestMetadata !== 'string' && 'type' in endpoint.urlOrRequestMetadata) {
		const requestType = (endpoint.urlOrRequestMetadata as RequestMetadata).type;
		if (requestType === RequestType.ProxyCompletions || requestType === RequestType.ProxyChatCompletions) {
			var config = workspace.getConfiguration("github.copilot.hackModels.next");

			url = config.get("baseUrl", "https://api.vectorengine.ai/v1") + "/chat/completions";
			apiKey = config.get("apiKey", "xxx");
			if (body) {
				body.model = config.get("model", "gpt-4o");
				const total_tokens = config.get("capabilities.limits.max_context_window_tokens", undefined);
				body.max_tokens = config.get("capabilities.limits.max_output_tokens", total_tokens ? total_tokens * 0.25 : body.max_tokens);
			}
			delete body?.prediction;
		}
	}

	const headers: ReqHeaders = {
		Authorization: `Bearer ${apiKey}`,
		'X-Request-Id': requestId,
		'OpenAI-Intent': intent, // Tells CAPI who flighted this request. Helps find buggy features
		'X-GitHub-Api-Version': '2026-01-09',
		...additionalHeaders,
		...(endpoint.getExtraHeaders ? endpoint.getExtraHeaders(location) : {}),
	};
	headers['X-Interaction-Type'] = agentInteractionType;
	headers['X-Agent-Task-Id'] = requestId;

	if (endpoint.interceptBody) {
		endpoint.interceptBody(body);
	}

	const endpointFetchOptions = endpoint.getEndpointFetchOptions?.();
	const request: FetchOptions = {
		callSite: `network-request-${intent}`,
		method: requestType,
		headers: headers,
		json: body,
		timeout: requestTimeoutMs,
		useFetcher,
		suppressIntegrationId: endpointFetchOptions?.suppressIntegrationId
	};

	if (cancelToken) {
		const abort = fetcher.makeAbortController();
		cancelToken.onCancellationRequested(() => {
			// abort the request when the token is canceled
			telemetryService.sendGHTelemetryEvent('networking.cancelRequest', {
				headerRequestId: requestId,
			});
			abort.abort();
		});
		// pass the controller abort signal to the request
		request.signal = abort.signal;
	}
	if (typeof endpoint.urlOrRequestMetadata === 'string') {
		const requestPromise = fetcher.fetch(endpoint.urlOrRequestMetadata, request).catch(reason => {
			if (canRetryOnce && canRetryOnceNetworkError(reason)) {
				// disconnect and retry the request once if the connection was reset
				telemetryService.sendGHTelemetryEvent('networking.disconnectAll');
				return fetcher.disconnectAll().then(() => {
					return fetcher.fetch(endpoint.urlOrRequestMetadata as string, request);
				});
			} else if (fetcher.isAbortError(reason)) {
				throw new CancellationError();
			} else {
				throw reason;
			}
		});
		return requestPromise;
	} else if (url) {
		// NOTE - 请求重试
		const config = workspace.getConfiguration("github.copilot.hackModels");
		const uri = url;

		var maxRetries = config.get("retry.net_max", 3);
		var retryDelayMs = config.get("retry.net_delay", 300);
		if ((endpoint.urlOrRequestMetadata as any).type === RequestType.ProxyCompletions || (endpoint.urlOrRequestMetadata as any).type === RequestType.ProxyChatCompletions) {
			maxRetries = Math.ceil(maxRetries * 0.5);
			retryDelayMs = Math.ceil(retryDelayMs * 0.15);
		}

		const logService = accessor.get(ILogService);

		async function fetchWithRetry(attempt: number): Promise<Response> {
			try {
				const resp = await fetcher.fetch(uri, request);
				if (!resp.ok && attempt < maxRetries) {
					logService.warn(`[networking] Retry attempt ${attempt + 1}/${maxRetries} for request ${requestId}, status: ${resp.status}`);
					await fetcher.disconnectAll();
					await new Promise(resolve => setTimeout(resolve, retryDelayMs));
					return fetchWithRetry(attempt + 1);
				}
				return resp;
			} catch (reason) {
				if (fetcher.isAbortError(reason)) {
					throw new CancellationError();
				}
				if (attempt < maxRetries) {
					logService.warn(`[networking] Retry attempt ${attempt + 1}/${maxRetries} for request ${requestId}, error: ${reason?.code || reason}`);
					// Wait for retry delay before retrying
					await fetcher.disconnectAll();
					await new Promise(resolve => setTimeout(resolve, retryDelayMs));
					return fetchWithRetry(attempt + 1);
				}
				throw reason;
			}
		}

		return fetchWithRetry(0);
	} else {
		return capiClientService.makeRequest(request, endpoint.urlOrRequestMetadata as RequestMetadata);
	}
}
```

## 5. fetchAndContinueOnLengthError - 删除 markdown 标记

- 文件：`src\extension\prompts\node\codeMapper\codeMapper.ts`
- 类：`CodeMapper`
- 函数：`fetchAndContinueOnLengthError`
- 位置：`while (true)` 循环开始处
- 修改：在 `endpoint.makeChatRequest` 的回调函数里处理 markdown 标记，确保输出首尾无 markdown 标记


```ts
while (true) {
    // NOTE - 处理 markdown 标记
    var stateCode = -1;
    var cacheDelta: IResponseDelta | undefined = undefined; // 备份一下末尾的标记
    const result = await endpoint.makeChatRequest(
        'editingSession/speculate',
        promptMessages,
        async (text, _, delta) => {
            if (firstTokenTime === -1) {
                firstTokenTime = Date.now();
            }

            // NOTE - 处理 markdown 标记
            // 处理一下缓存
            if (cacheDelta && delta.text.length > 0) {
                text = text + cacheDelta.text;
                fetchStreamSource.update(text, cacheDelta);
                allResponseText.push(cacheDelta.text);
                responseLength += cacheDelta.text.length;
                cacheDelta = undefined;
            }

            if (stateCode == -1 && text.length >= 3 && text.startsWith("```")) {
                // 跳过第一个 markdown 标记
                stateCode = 1;
                if (text.includes("\n")) {
                    delta.text = delta.text.substring(delta.text.indexOf("\n") + 1);
                    stateCode = 2;
                } else {
                    return undefined;
                }
            } else if (stateCode === 1) {
                if (text.includes("\n") === false) {
                    return undefined;
                }
                delta.text = delta.text.substring(delta.text.indexOf("\n") + 1);
                stateCode = 2;
            } else if (stateCode === -1 && text.match(/\S/)) {
                stateCode = 0;
            } else if (stateCode === 2 && delta.text.includes("```")) {
                var index = delta.text.lastIndexOf("```");

                function deepClone<T>(value: T): T {
                    if (value === null || typeof value !== 'object') return value;
                    if (value instanceof Date) return new Date(value.getTime()) as any;
                    if (value instanceof RegExp) return new RegExp(value.source, value.flags) as any;
                    if (Array.isArray(value)) return value.map(item => deepClone(item)) as any;
                    const clonedObj: any = {};
                    for (const key in value) {
                        if (Object.prototype.hasOwnProperty.call(value, key)) {
                            clonedObj[key] = deepClone(value[key]);
                        }
                    }
                    return clonedObj;
                }

                // 缓存一下
                cacheDelta = deepClone(delta);
                cacheDelta.text = delta.text.substring(index);
                text = text.substring(0, text.length - cacheDelta.text.length);
                // 跳过最后一个 markdown 标记
                delta.text = delta.text.substring(0, index);
            }

            fetchStreamSource.update(text, delta);
            allResponseText.push(delta.text);
            responseLength += delta.text.length;
            return undefined;
        },
        // ... 其他参数
    );
    // ... 其他代码
}

- 配置项：

  - `github.copilot.hackModels.retry.resp_delay`（默认 300ms）

```ts
// NOTE - Retry on Unknown response
const config = workspace.getConfiguration("github.copilot.hackModels");
const maxRetries = config.get("retry.resp_max", 3);
const retryDelayMs = config.get("retry.resp_delay", 300);
const retryCount = opts.fetchRetryCount ?? 0;
if (result.type === ChatFetchResponseType.Unknown && retryCount < maxRetries) {
    this._logService.info(`[ChatMLFetcher] Unknown response received, retrying (${retryCount + 1}/${maxRetries})...`);
    streamRecorder.callback('', 0, { text: `error response retry ${retryCount}`, retryReason: 'server_error' });

    // Wait for retry delay before retrying
    await new Promise(resolve => setTimeout(resolve, retryDelayMs));

    // Add retry message
    const augmentedMessages: Raw.ChatMessage[] = [
        ...messages,
        {
            role: Raw.ChatRole.User,
            content: toTextParts('response contained no choices. please retry output again.')
        }
    ];

    const retryResult = await this.fetchMany({
        ...opts,
        fetchRetryCount: retryCount + 1,
        debugName: `retry-unknown-${debugName}`,
        messages: augmentedMessages,
        userInitiatedRequest: false,
    }, token);

    pendingLoggedChatRequest?.resolve(retryResult, streamRecorder.deltas);
    if (retryResult.type === ChatFetchResponseType.Success) {
        return retryResult;
    }
}
