```ts
function addCommentByLanguageId(text: string, languageId: string): string {
	const lang = languageId.toLowerCase();

	const commentMap: Record<string, { line?: string; block?: [string, string] }> = {
		javascript: { line: '// ' },
		typescript: { line: '// ' },
		java: { line: '// ' },
		cpp: { line: '// ' },
		c: { line: '// ' },
		csharp: { line: '// ' },
		go: { line: '// ' },
		rust: { line: '// ' },
		php: { line: '// ' },
		swift: { line: '// ' },
		kotlin: { line: '// ' },
		dart: { line: '// ' },
		scala: { line: '// ' },
		zig: { line: '// ' },
		python: { line: '# ' },
		ruby: { line: '# ' },
		perl: { line: '# ' },
		yaml: { line: '# ' },
		makefile: { line: '# ' },
		powershell: { line: '# ' },
		dockerfile: { line: '# ' },
		toml: { line: '# ' },
		sql: { line: '-- ' },
		lua: { line: '-- ' },
		haskell: { line: '-- ' },
		ada: { line: '-- ' },
		lisp: { line: '; ' },
		scheme: { line: '; ' },
		clojure: { line: '; ' },
		html: { block: ['<!-- ', ' -->'] },
		xml: { block: ['<!-- ', ' -->'] },
		css: { block: ['/* ', ' */'] },
		markdown: { block: ['<!-- ', ' -->'] },
	};

	const config = commentMap[lang];
	if (!config) {
		return text;
	}

	if (config.line) {
		const lines = text.split(/\r?\n/);
		const commentedLines = lines.map(line => config.line! + line);
		return commentedLines.join('\n');
	}

	if (config.block) {
		const [start, end] = config.block;
		return `${start}${text}${end}`;
	}

	return text;
}
```

```ts
async fetchAndStreamCompletions(
	params: CompletionParams,
	baseTelemetryData: TelemetryWithExp,
	finishedCb: FinishedCallback,
	cancel?: ICancellationToken
): Promise<CompletionResults | CompletionError> {
	// ... error checks, base request construction ...

	// Read inline config for hack models
	const inline_config = workspace.getConfiguration("github.copilot.hackModels.inline");
	const inline_uri = inline_config.get("baseUrl", "https://steam/v1") as string;
	const inline_mode = inline_config.get("mode", "chat") as string;
	const inline_token = inline_config.get("apiKey", "") as string;
	const inline_stops = inline_config.get("stop", []) as string[];
	const inline_interval = inline_config.get("capabilities.limits.delay", 200) as number;
	const inline_model = inline_config.get("model", "") as string;

	const model_config = workspace.getConfiguration("github.copilot.hackModels");
	const frequency_penalty = model_config.get("frequency_penalty", undefined) as number | undefined;
	const presence_penalty = model_config.get("presence_penalty", undefined) as number | undefined;

	// ... variable aliases for prompt, engineModelId, ourRequestId, telemetryExp, uiKind, headers ...

	// Determine URI based on inline config or fallback to original
	let uri = inline_uri.length > 0
		? inline_uri
		: this.instantiationService.invokeFunction(getProxyEngineUrl, copilotToken, engineModelId, endpoint);

	// Build the request with inline config modifications
	const req = request as Partial<CompletionRequest>;

	// Apply penalty settings
	if (frequency_penalty !== undefined) {
		req.frequency_penalty = frequency_penalty;
	}
	if (presence_penalty !== undefined) {
		req.presence_penalty = presence_penalty;
	}

	// Merge stop sequences
	if (req.stop) {
		req.stop.push(...inline_stops);
	} else {
		req.stop = [...inline_stops];
	}

	// Set model from inline config
	if (inline_model.length > 0) {
		req.model = inline_model;
	}

	// Process extra context with language-based commenting
	if (req.extra) {
		if (req.extra.context && req.extra.context.length > 0 && req.extra.language && req.prompt) {
			req.prompt = addCommentByLanguageId(req.extra.context.join('\n'), req.extra.language) + '\n\n' + req.prompt;
		}
	}

	// Handle mode-specific transformations
	if (inline_mode === "api") {
		uri = uri + "/completions";
		delete req.code_annotations;
		delete req.extra;
	} else if (inline_mode === "fim") {
		uri = uri + "/completions";
		const fim_begin_prefix = inline_config.get("fim.begin_prefix", "<|fim_prefix|>") as string;
		const fim_end_prefix = inline_config.get("fim.end_prefix", "") as string;
		const fim_begin_suffix = inline_config.get("fim.begin_suffix", "<|fim_suffix|>\n") as string;
		const fim_end_suffix = inline_config.get("fim.end_suffix", "") as string;
		const fim_begin_mid = inline_config.get("fim.begin_mid", "<|fim_middle|>") as string;

		req.prompt = `${fim_begin_prefix}${req.prompt}${fim_end_prefix}${fim_begin_suffix}${req.suffix}${fim_end_suffix}${fim_begin_mid}`;

		const fim_end_mid = inline_config.get("fim.end_mid", "") as string;
		if (fim_end_mid.length > 0 && req.stop) {
			req.stop.push(fim_end_mid);
		}

		delete req.suffix;
		delete req.code_annotations;
		delete req.extra;
	}

	// ... telemetry setup, headers construction ...

	// Use inline token if provided, otherwise use copilot token
	const token = inline_token.length > 0 ? inline_token : copilotToken.token;

	// Rate-limiting network call
	const makeRequest = async () => {
		const response = await this.fetchService.fetch(
			uri,
			token,
			req as Completions.ModelParams,
			ourRequestId,
			cancelToken,
			fullHeaders,
		);
		lastRequestTime = Date.now();
		return response;
	};

	let res = await new Promise<ReturnType<typeof this.fetchService.fetch>>(async (resolve, reject) => {
		const now = Date.now();
		const timeSinceLastRequest = now - lastRequestTime;
		const waitTime = Math.max(0, inline_interval - timeSinceLastRequest);

		const timeoutId = setTimeout(async () => {
			try {
				const response = makeRequest();
				resolve(response);
			} catch (e) {
				lastRequestTime = Date.now();
				reject(e);
			}
		}, waitTime);

		if (lastTimeoutId) {
			clearTimeout(lastTimeoutId);
		}
		lastTimeoutId = timeoutId;
	}).catch(error => {
		throw error;
	}).then(async response => {
		// Retry on interrupted network error
		const resp = await response;
		if (resp.isError() && resp.err instanceof Completions.Unexpected && isInterruptedNetworkError(resp.err.error)) {
			this.instantiationService.invokeFunction(telemetry, 'networking.disconnectAll');
			await this.fetchService.disconnectAll();
			return makeRequest();
		}
		return response;
	});

	// ... error handling, stream conversion ...
}
```
